## AppUser Entity

```csharp
using Microsoft.AspNetCore.Identity;
using System.ComponentModel.DataAnnotations;
using System.ComponentModel.DataAnnotations.Schema;

public class AppUser : IdentityUser<Guid>
{
    [Required]
    [MaxLength(100)]
    public string Name { get; set; } = string.Empty;

    [Required]
    [MaxLength(100)]
    public string LastName { get; set; } = string.Empty;

    [NotMapped]
    public string FullName => $"{Name} {LastName}";

    public string? Avatar { get; set; }

    public bool IsApiUser { get; set; }

    public UserStatus UserStatus { get; set; } = UserStatus.AwaitingApproval;

    // Extensible properties (Option 1: JSON Column for EF Core 7+)
    public Dictionary<string, object> ExtendedProperties { get; set; } = new();

    // Navigation properties
    public ICollection<UserProperty> UserProperties { get; set; } = new List<UserProperty>();
    public ICollection<AppUserRole> UserRoles { get; set; } = new List<AppUserRole>();
    public ICollection<AppUserClaim> UserClaims { get; set; } = new List<AppUserClaim>();
    public ICollection<AppUserLogin> UserLogins { get; set; } = new List<AppUserLogin>();
    public ICollection<AppUserToken> UserTokens { get; set; } = new List<AppUserToken>();
}
```

## DbContext Configuration

```csharp
public class ApplicationDbContext : IdentityDbContext<AppUser, AppRole, Guid,
    AppUserRole, AppUserClaim, AppUserLogin, AppRoleClaim, AppUserToken>
{
    public ApplicationDbContext(DbContextOptions<ApplicationDbContext> options)
        : base(options)
    {
    }

    public DbSet<Module> Modules { get; set; }
    public DbSet<ModuleClaim> ModuleClaims { get; set; }
    public DbSet<GeneratedId> GeneratedIds { get; set; }
    public DbSet<UserProperty> UserProperties { get; set; }

    protected override void OnModelCreating(ModelBuilder builder)
    {
        base.OnModelCreating(builder);

        // AppUser configuration
        builder.Entity<AppUser>(entity =>
        {
            entity.ToTable("Users");
            entity.Property(e => e.Name).IsRequired().HasMaxLength(100);
            entity.Property(e => e.LastName).IsRequired().HasMaxLength(100);
            entity.Property(e => e.Avatar).HasMaxLength(500);
            entity.Property(e => e.UserStatus).HasConversion<byte>();
        });

        // AppRole configuration
        builder.Entity<AppRole>(entity =>
        {
            entity.ToTable("Roles");
            entity.Property(e => e.Description).IsRequired().HasMaxLength(500);
        });

        // Module configuration
        builder.Entity<Module>(entity =>
        {
            entity.ToTable("Modules");
            entity.HasKey(e => e.Id);
            entity.Property(e => e.Name).IsRequired().HasMaxLength(100);
            entity.Property(e => e.DisplayName).HasMaxLength(200);
            entity.Property(e => e.Description).IsRequired().HasMaxLength(500);
        });

        // ModuleClaim configuration
        builder.Entity<ModuleClaim>(entity =>
        {
            entity.ToTable("ModuleClaims");
            entity.HasKey(e => e.Id);
            entity.Property(e => e.ModuleId).IsRequired();
            entity.Property(e => e.Name).HasMaxLength(100);
            entity.Property(e => e.DisplayName).HasMaxLength(200);
            entity.Property(e => e.Description).HasMaxLength(500);

            entity.HasOne(e => e.Module)
                .WithMany(m => m.Claims)
                .HasForeignKey(e => e.ModuleId)
                .OnDelete(DeleteBehavior.Cascade);
        });

        // GeneratedId configuration
        builder.Entity<GeneratedId>(entity =>
        {
            entity.ToTable("GeneratedIds");
            entity.HasKey(e => e.Id);
            entity.Property(e => e.ModuleId).IsRequired();
            entity.Property(e => e.Name).IsRequired().HasMaxLength(100);
            entity.Property(e => e.Description).IsRequired().HasMaxLength(500);
            entity.Property(e => e.Prefix).IsRequired().HasMaxLength(20);

            entity.HasOne(e => e.Module)
                .WithMany(m => m.GeneratedIds)
                .HasForeignKey(e => e.ModuleId)
                .OnDelete(DeleteBehavior.Cascade);
        });

        // UserProperty configuration
        builder.Entity<UserProperty>(entity =>
        {
            entity.ToTable("UserProperties");
            entity.HasKey(e => e.Id);
            entity.Property(e => e.UserId).IsRequired();
            entity.Property(e => e.Key).IsRequired().HasMaxLength(100);
            entity.Property(e => e.Value).IsRequired();
            entity.Property(e => e.Type).HasConversion<byte>();

            entity.HasOne(e => e.User)
                .WithMany(u => u.UserProperties)
                .HasForeignKey(e => e.UserId)
                .OnDelete(DeleteBehavior.Cascade);
        });
    }
}
```

## Avatar Storage Helper

```csharp
public static class AvatarHelper
{
    private static readonly string AvatarPath = Path.Combine(Directory.GetCurrentDirectory(), "wwwroot", "Upload", "Avatars");

    static AvatarHelper()
    {
        if (!Directory.Exists(AvatarPath))
        {
            Directory.CreateDirectory(AvatarPath);
        }
    }

    public static async Task<string> SaveAvatarAsync(IFormFile file)
    {
        var extension = Path.GetExtension(file.FileName);
        var fileName = $"{Guid.NewGuid()}{extension}";
        var fullPath = Path.Combine(AvatarPath, fileName);

        using (var stream = new FileStream(fullPath, FileMode.Create))
        {
            await file.CopyToAsync(stream);
        }

        return $"/Upload/Avatars/{fileName}";
    }

    public static void DeleteAvatar(string? avatarPath)
    {
        if (string.IsNullOrEmpty(avatarPath)) return;

        var fullPath = Path.Combine(Directory.GetCurrentDirectory(), "wwwroot", avatarPath.TrimStart('/'));
        if (File.Exists(fullPath))
        {
            File.Delete(fullPath);
        }
    }
}
```
